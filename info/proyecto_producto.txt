class ProductoController(object): #Creo la clase ProductoController, atento a object
        
    def __init__(self): #Creo el método constructor, atento a self
        self.model = ProductoModel() #Creo una instancia del modelo
        self.view = ProductoView() #Creo una instancia de la vista

    def ver(self):  #Llamo a la vista 
        self.model.producto_id = 1
        self.model.select()

        self.view.ver(self.model)
